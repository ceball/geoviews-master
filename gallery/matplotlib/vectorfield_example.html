Define data ¶

In [2]: 
xs, ys, U, V, crs = arrows.sample_data()
mag = np.sqrt(U**2 + V**2)
angle = (np.pi/2.) - np.arctan2(U/mag, V/mag)
tiles = gv.WMTS('https://maps.wikimedia.org/osm-intl/{Z}/{X}/{Y}@2x.png')
vectorfield = gv.VectorField((xs, ys, angle, mag), crs=crs)
